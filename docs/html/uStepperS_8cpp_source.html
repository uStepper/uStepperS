257 {
258  static float oldTargetPosition;
259  static float oldEncoderPosition;
260  static float encoderPositionChange;
261  static float targetPositionChange;
262  float encoderPosition = ((float)this->encoder.angleMoved*ENCODERDATATOSTEP);
263  static float internalStall = 0.0;
264 
265  encoderPositionChange *= 0.99;
266  encoderPositionChange += 0.01*(oldEncoderPosition - encoderPosition);
267  oldEncoderPosition = encoderPosition;
268 
269  targetPositionChange *= 0.99;
270  targetPositionChange += 0.01*(oldTargetPosition - stepsMoved);
271  oldTargetPosition = stepsMoved;
272 
273  if(abs(encoderPositionChange) < abs(targetPositionChange)*0.5)
274  {
275  internalStall *= this->stallSensitivity;
276  internalStall += 1.0-this->stallSensitivity;
277  }
278  else
279  {
280  internalStall *= this->stallSensitivity;
281  }
282 
283  if(internalStall >= 0.95) //3 timeconstants

533

539  deltaAngle += 65535;
540  }
541  else if(deltaAngle > 32768)
542  {
543  deltaAngle -= 65535;

622 float uStepperS::pid(float error)
623 {
624  float u;
625  float limit = abs(this->currentPidSpeed) + 150000.0;
626  static float integral;
627  static bool integralReset = 0;
628  static float errorOld, differential = 0.0;
629 
630  this->currentPidError = error;
631 
632  u = error*this->pTerm;
633 
634  if(u > 0.0)
635  {
636  if(u > limit)
637  {
638  u = limit;
639  }
640  }
641  else if(u < 0.0)
642  {
643  if(u < -limit)
644  {
645  u = -limit;
646  }
647  }
648 
649  integral += error*this->iTerm;
650 
651  if(integral > 3000000.0)
652  {
653  integral = 3000000.0;
654  }
655  else if(integral < -3000000.0)
656  {
657  integral = -3000000.0;
658  }
659 
660  if(error > -100 && error < 100)
661  {
662  if(!integralReset)
663  {
664  integralReset = 1;
665  integral = 0;
666  }
667  }
668  else
669  {
670  integralReset = 0;
671  }
672 
673  u += integral;
674 
675  differential *= 0.0;
676  differential += 1.0*((error - errorOld)*this->dTerm);
677 
678  errorOld = error;
679 
680  u += differential;
681 
682  u *= this->stepsPerSecondToRPM;
683  this->setRPM(u);
